# 五：Mybatis各种查询功能

# 1、查询一个实体类对象

/**
* 根据id查询用户信息
*/
User getUserById(Long id);

<!-- User getUserById(Long id); -->
<select id="getUserById" resultType="com.suji.mybatis.domain.User">
    SELECT * FROM user WHERE id = #{id}
</select>

# 2、查询一个list集合

/**
 * 根据性别查询用户信息
 */
List<User> getUserBySex(String sex);

<!-- List<User> getUserBySex(String sex); -->
<select id="getUserBySex" resultType="com.suji.mybatis.domain.User">
    SELECT * FROM user WHERE sex = #{sex}
</select>

# 3、查询单个数据

/**
 * 查询用户信息的总记录数
 */
Integer countAllUser();

<!-- Integer countAllUser(); -->
<select id="countAllUser" resultType="integer">
    SELECT COUNT(*) FROM user
</select>

备注： count(*) 不一定等于 count(字段)【count的字段如果为null是不会计数的】

mybatis中设置了默认的类型别名

# 4、查询一条数据为map集合

/**
 * 根据id用户信息为一个map集合
 */
Map<String, Object> getUserByIdToMap(Long id);

<!-- Map<String, Object> getUserByIdToMap(Long id); -->
<select id="getUserByIdToMap" resultType="map">
    SELECT * FROM user WHERE id = #{id}
</select>
<!--查询结果：{password=admin123, sex=女, id=1, email=1044545111@qq.com, username=admin} -->

# 5、查询多条数据为map集合

/**
 * 根据所有用户信息为一个map集合
 */
List<Map<String, Object>> getAllUserByIdToMap();

<!-- 根据所有用户信息为一个map集合 -->
<select id="getAllUserByIdToMap" resultType="map">
    SELECT * FROM user
</select>

@MapKey 注解

设置map集合的键。把我们查询出来的字段设置为键，查询出来的map集合作为值（键重复会把值覆盖掉）

/**
 * 根据所有用户信息为一个map集合
 */
@MapKey("id")
Map<String, Object> getAllUserByIdToMap2();

<!-- @MapKey("id")
     Map<String, Object> getAllUserByIdToMap2(); -->
<select id="getAllUserByIdToMap2"  resultType="map">
    SELECT * FROM user
</select>

<!-- 输出结果：
{
    1={password=admin123, sex=女, id=1, email=1044545111@qq.com, username=admin}, 
    4={password=admin123, sex=女, id=4, email=1044545111@qq.com, username=admin1}}
-->
